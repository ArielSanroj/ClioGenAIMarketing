// src/components/common/Button.jsx
import React from 'react';
import { COLORS } from '@/constants/theme';

export const Button = ({ 
  children, 
  variant = 'primary', 
  highlighted = false,
  ...props 
}) => {
  const getStyles = () => {
    switch (variant) {
      case 'primary':
        return {
          backgroundColor: COLORS.primary,
          color: 'white',
        };
      case 'action':
        return {
          backgroundColor: highlighted ? COLORS.highlight : 'white',
          borderColor: highlighted ? COLORS.highlightBorder : '#E5E7EB',
        };
      default:
        return {};
    }
  };

  return (
    <button
      className="px-6 py-3 rounded-lg border transition-colors duration-200"
      style={getStyles()}
      {...props}
    >
      {children}
    </button>
  );
};
